public:
    int *findTwoElement(int *arr, int n) {
        // code here
        int freq[n+1] ={0};
        for(int i=0; i<n;i++){
            freq[arr[i]]++;
        }
        int *ans = new int[2];
        for(int i =0;i<n+1;i++){
            if(freq[i] == 0){
                ans[1] = i;
            }
            if(freq[i] > 1){
                ans[0] = i;
            }
        }
        return ans;
    }
};
