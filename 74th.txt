int minFlips (string S)
{
    // your code here
    int n = S.length();
    int counter1 = 0, counter2 = 0;
    int mini = INT_MAX;
    for(int i=0; i<n;i++){
        if((i%2 == 0 && S[i] == '0') || (i%2 != 0 &&S[i] == '1')){
            counter1++;
        }
        if((i%2 == 0 && S[i] == '1') || (i%2 != 0 &&S[i] == '0')){
            counter2++;
        }
    }
    mini = min(counter1,counter2);
    return mini;
}
