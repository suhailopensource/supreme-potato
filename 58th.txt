class Solution {
public:
    int maxSubStr(string str){
        //Write your code here
        int x=0;
        int y=0;
        int cnt=0;
        for(int i=0;i<str.length();i++){
            if(str[i]=='0'){
                x++;}
            else
                y++;
            if(x==y) cnt++; 
            
        }
        if(x!=y) return -1;
        else return cnt;
    }
};
