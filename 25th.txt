#include<bits/stdc++.h>
using namespace std;

void func(int arr[], int n, int k){
    unordered_map<int,int> m;
    int x = n/k;
    for(int i=0; i<n;i++){
        m[arr[i]]++;
    }
    for(auto i : m){
        if(i.second > x){
            cout << i.first << " ";
        }
    }
}

int main(){
     int n;
     cin >> n;
     int k;
     cin>> k;
     int arr[n];
     for(int i=0; i<n;i++){
        cin >> arr[i];
     }
     func(arr, n, k);

}
