{
public:
    vector<int> singleNumber(vector<int> nums) 
    {
        // Code here.
        vector<int> ans;
        sort(nums.begin(),nums.end());
        for(int i =0; i< nums.size();){
            int x = nums[i]^nums[i+1];
            if(x==0){
                i+=2;
            }else{
                ans.push_back(nums[i]);
                i++;
            }
        }
        return ans;
    }
};
