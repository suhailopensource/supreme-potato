class Solution {
  public:
    vector<vector<int>> sortedMatrix(int n, vector<vector<int>> mat) {
        // code here
        vector<int> v;
        int i,j,k = 0;
        for(i =0; i<n;i++){
            for(j = 0; j<n;j++){
                v.push_back(mat[i][j]);
            }
        }
       sort(v.begin(),v.end());
       for(i =0; i<n;i++){
           for(j = 0; j<n;j++){
               mat[i][j]  = v[k++];
           }
       }
       return mat;
    }
};
