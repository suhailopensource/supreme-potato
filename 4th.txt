void sort012(int a[], int n)
    {
        // code here 
     int low = 0, mid = 0, high = n-1;
     
     while(mid<=high){
         int x = a[mid];
         if(x==0){
             swap(a[low],a[mid]); 
             low++;
             mid++;
         }
         else if(x==1){
             mid++;
         }else{
             swap(a[high],a[mid]);
             high--;
            }
        }
    }
