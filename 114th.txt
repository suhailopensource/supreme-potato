class Solution{
    public:
    //Function to count subarrays with sum equal to 0.
    ll findSubarray(vector<ll> arr, int n ) 
    {
        ll ans=0;
        ll sum=0;
        unordered_map<ll,ll>mp;
        for(int i=0;i<n;i++)
        {
            sum+=arr[i];
            if(sum==0)ans++;
            if(mp.find(sum)!=mp.end())ans+=mp[sum];
            mp[sum]++;
        }
        return ans;
    }
};
