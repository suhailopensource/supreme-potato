#include <bits/stdc++.h>
using namespace std;

void printdups(string str){
   map<char,int> mp;
    for(int i = 0; i< str.length();i++){
        mp[str[i]]++;
    }
    for(auto p : mp){
        if(p.second > 1){
            cout << p.first << " " << p.second << endl;
        }
    }

}

int main(){
   string str;
   cin >> str;
    printdups(str);
    return 0;
}
