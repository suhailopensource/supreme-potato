
class Solution{
public:
    int isStackPermutation(int N,vector<int> &A,vector<int> &B){
        stack<int> st;
        for(int i =0,j =0; i<N;i++){
            st.push(A[i]);
            while(!st.empty() && st.top()==B[j]){
                st.pop();
                j++;
            }
        }
        return st.empty();
    }
};
